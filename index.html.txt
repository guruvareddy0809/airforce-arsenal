<script>
    const toys = [
        { name: "Teddy Bear", price: 499, image: "https://via.placeholder.com/200x150?text=Teddy" },
        { name: "Toy Car", price: 299, image: "https://via.placeholder.com/200x150?text=Car" },
        { name: "Doll", price: 399, image: "https://via.placeholder.com/200x150?text=Doll" },
        { name: "Lego Set", price: 899, image: "https://via.placeholder.com/200x150?text=Lego" },
        { name: "Robot Toy", price: 699, image: "https://via.placeholder.com/200x150?text=Robot" }
    ];

    let cartCount = 0;

    function displayToys(toyList) {
        const container = document.getElementById("toyContainer");
        container.innerHTML = "";

        toyList.forEach(toy => {
            const card = document.createElement("div");
            card.className = "toy-card";

            card.innerHTML = `
                <img src="${toy.image}" alt="${toy.name}">
                <h3>${toy.name}</h3>
                <p>â‚¹${toy.price}</p>
                <button onclick="addToCart()">Add to Cart</button>
            `;
            container.appendChild(card);
        });
    }

    function addToCart() {
        cartCount++;
        document.getElementById("cartCount").innerText = cartCount;
    }

    function searchToys() {
        const searchValue = document.getElementById("searchInput").value.toLowerCase();
        const filteredToys = toys.filter(toy =>
            toy.name.toLowerCase().includes(searchValue)
        );
        displayToys(filteredToys);
    }

    // Initial load
    displayToys(toys);
</script>
